md to tex

pandoc introductory_essay_2019.tex --include-in-header=header.tex -R --csl=ecology-letters.csl -o test.pdf -s -S --latex-engine=xelatex --bibliography=citationLibPrg2019.bib --variable documentclass=scrartcl
